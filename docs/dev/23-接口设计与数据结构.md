# 23. 接口设计与数据结构

## 23.1 核心接口定义

### 23.1.1 IAgent 接口

IAgent 是 MSRA Agent 框架的核心接口，定义了所有 Agent 必须实现的方法。

```python
from abc import ABC, abstractmethod
from typing import Dict, Any, List, Optional
from enum import Enum

class AgentStatus(Enum):
    """Agent 状态枚举"""
    IDLE = "idle"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    WAITING = "waiting"
    SKIPPED = "skipped"

class AgentCapability(Enum):
    """Agent 能力枚举"""
    DATA_CLEANING = "data_cleaning"
    ANALYSIS_PLANNING = "analysis_planning"
    ANALYSIS_EXECUTION = "analysis_execution"
    REPORT_GENERATION = "report_generation"
    PAPER_WRITING = "paper_writing"
    QUALITY_REVIEW = "quality_review"
    CALIBRATION = "calibration"

class IAgent(ABC):
    """Agent 核心接口"""
    
    @property
    @abstractmethod
    def agent_id(self) -> str:
        """Agent 唯一标识"""
        pass
    
    @property
    @abstractmethod
    def agent_name(self) -> str:
        """Agent 名称"""
        pass
    
    @property
    @abstractmethod
    def capabilities(self) -> List[AgentCapability]:
        """Agent 能力列表"""
        pass
    
    @property
    @abstractmethod
    def status(self) -> AgentStatus:
        """当前状态"""
        pass
    
    @abstractmethod
    def initialize(self, config: Dict[str, Any]) -> bool:
        """
        初始化 Agent
        
        Args:
            config: 配置参数
            
        Returns:
            bool: 初始化是否成功
        """
        pass
    
    @abstractmethod
    def execute(self, input_data: Dict[str, Any]) -> Dict[str, Any]:
        """
        执行 Agent 任务
        
        Args:
            input_data: 输入数据
            
        Returns:
            Dict[str, Any]: 执行结果
        """
        pass
    
    @abstractmethod
    def validate_input(self, input_data: Dict[str, Any]) -> bool:
        """
        验证输入数据
        
        Args:
            input_data: 输入数据
            
        Returns:
            bool: 验证是否通过
        """
        pass
    
    @abstractmethod
    def validate_output(self, output_data: Dict[str, Any]) -> bool:
        """
        验证输出数据
        
        Args:
            output_data: 输出数据
            
        Returns:
            bool: 验证是否通过
        """
        pass
    
    @abstractmethod
    def get_handoff_data(self) -> Dict[str, Any]:
        """
        获取交接数据
        
        Returns:
            Dict[str, Any]: 交接数据
        """
        pass
    
    @abstractmethod
    def accept_handoff(self, handoff_data: Dict[str, Any]) -> bool:
        """
        接收交接数据
        
        Args:
            handoff_data: 交接数据
            
        Returns:
            bool: 接收是否成功
        """
        pass
    
    def get_progress(self) -> float:
        """
        获取执行进度
        
        Returns:
            float: 进度百分比 (0.0 - 1.0)
        """
        return 0.0
    
    def get_diagnostics(self) -> Dict[str, Any]:
        """
        获取诊断信息
        
        Returns:
            Dict[str, Any]: 诊断信息
        """
        return {}
    
    def cleanup(self) -> None:
        """清理资源"""
        pass
```

### 23.1.2 Handoff 接口

Handoff 接口定义了跨阶段数据交接的规范。

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Dict, Any, List, Optional

@dataclass
class HandoffMetadata:
    """Handoff 元数据"""
    handoff_id: str
    source_agent: str
    target_agent: str
    timestamp: datetime
    version: str
    checksum: str

@dataclass
class HandoffData:
    """Handoff 数据结构"""
    metadata: HandoffMetadata
    artifacts: Dict[str, Any]
    quality_gates: Dict[str, bool]
    validation_results: Dict[str, Any]
    
    def validate(self) -> bool:
        """验证 Handoff 数据完整性"""
        # 检查必需字段
        if not self.metadata.handoff_id:
            return False
        
        # 检查质量门闸
        if not all(self.quality_gates.values()):
            return False
        
        return True

class HandoffManager:
    """Handoff 管理器"""
    
    def create_handoff(
        self,
        source_agent: str,
        target_agent: str,
        artifacts: Dict[str, Any],
        quality_gates: Dict[str, bool]
    ) -> HandoffData:
        """创建 Handoff"""
        metadata = HandoffMetadata(
            handoff_id=self._generate_id(),
            source_agent=source_agent,
            target_agent=target_agent,
            timestamp=datetime.now(),
            version="1.0",
            checksum=self._calculate_checksum(artifacts)
        )
        
        return HandoffData(
            metadata=metadata,
            artifacts=artifacts,
            quality_gates=quality_gates,
            validation_results={}
        )
    
    def transfer(self, handoff: HandoffData) -> bool:
        """执行 Handoff 传输"""
        if not handoff.validate():
            return False
        
        # 保存 Handoff 数据
        self._save_handoff(handoff)
        
        # 通知目标 Agent
        self._notify_target(handoff)
        
        return True
```

### 23.1.3 AgentMessage 接口

AgentMessage 定义了 Agent 间通信的消息格式。

```python
from dataclasses import dataclass
from datetime import datetime
from enum import Enum
from typing import Dict, Any, Optional

class MessageType(Enum):
    """消息类型"""
    REQUEST = "request"
    RESPONSE = "response"
    NOTIFICATION = "notification"
    ERROR = "error"
    HEARTBEAT = "heartbeat"

class MessagePriority(Enum):
    """消息优先级"""
    LOW = 0
    NORMAL = 1
    HIGH = 2
    CRITICAL = 3

@dataclass
class AgentMessage:
    """Agent 消息结构"""
    message_id: str
    message_type: MessageType
    priority: MessagePriority
    source_agent: str
    target_agent: str
    timestamp: datetime
    payload: Dict[str, Any]
    correlation_id: Optional[str] = None
    reply_to: Optional[str] = None
    ttl: Optional[int] = None  # 生存时间（秒）
    
    def is_expired(self) -> bool:
        """检查消息是否过期"""
        if self.ttl is None:
            return False
        elapsed = (datetime.now() - self.timestamp).total_seconds()
        return elapsed > self.ttl
    
    def create_reply(self, payload: Dict[str, Any]) -> 'AgentMessage':
        """创建回复消息"""
        return AgentMessage(
            message_id=self._generate_id(),
            message_type=MessageType.RESPONSE,
            priority=self.priority,
            source_agent=self.target_agent,
            target_agent=self.source_agent,
            timestamp=datetime.now(),
            payload=payload,
            correlation_id=self.message_id
        )

class MessageBus:
    """消息总线"""
    
    def __init__(self):
        self._subscribers: Dict[str, List[callable]] = {}
        self._message_queue: List[AgentMessage] = []
    
    def subscribe(self, agent_id: str, callback: callable) -> None:
        """订阅消息"""
        if agent_id not in self._subscribers:
            self._subscribers[agent_id] = []
        self._subscribers[agent_id].append(callback)
    
    def publish(self, message: AgentMessage) -> None:
        """发布消息"""
        if message.is_expired():
            return
        
        if message.target_agent in self._subscribers:
            for callback in self._subscribers[message.target_agent]:
                callback(message)
    
    def request(
        self,
        source: str,
        target: str,
        payload: Dict[str, Any],
        timeout: int = 30
    ) -> Optional[AgentMessage]:
        """发送请求并等待响应"""
        message = AgentMessage(
            message_id=self._generate_id(),
            message_type=MessageType.REQUEST,
            priority=MessagePriority.NORMAL,
            source_agent=source,
            target_agent=target,
            timestamp=datetime.now(),
            payload=payload,
            ttl=timeout
        )
        
        self.publish(message)
        return self._wait_for_response(message.message_id, timeout)
```

### 23.1.4 ConflictReport 接口

ConflictReport 定义了冲突检测和解决的数据结构。

```python
from dataclasses import dataclass
from enum import Enum
from typing import Dict, Any, List, Optional

class ConflictLevel(Enum):
    """冲突级别"""
    INFO = "info"
    WARNING = "warning"
    ERROR = "error"
    CRITICAL = "critical"

class ConflictType(Enum):
    """冲突类型"""
    DATA_INCONSISTENCY = "data_inconsistency"
    METHOD_MISMATCH = "method_mismatch"
    RESULT_DIVERGENCE = "result_divergence"
    CONSTRAINT_VIOLATION = "constraint_violation"

@dataclass
class ConflictDetail:
    """冲突详情"""
    conflict_type: ConflictType
    description: str
    affected_artifacts: List[str]
    evidence: Dict[str, Any]
    suggested_resolution: Optional[str] = None

@dataclass
class ConflictReport:
    """冲突报告"""
    report_id: str
    timestamp: datetime
    conflict_level: ConflictLevel
    conflicts: List[ConflictDetail]
    source_agents: List[str]
    resolution_status: str  # pending, resolved, escalated
    resolution_notes: Optional[str] = None
    
    def has_critical_conflicts(self) -> bool:
        """检查是否有严重冲突"""
        return any(
            c.conflict_type in [ConflictType.DATA_INCONSISTENCY, ConflictType.CONSTRAINT_VIOLATION]
            for c in self.conflicts
        )
    
    def get_resolution_suggestions(self) -> List[str]:
        """获取解决建议"""
        return [
            c.suggested_resolution
            for c in self.conflicts
            if c.suggested_resolution
        ]

class ConflictResolver:
    """冲突解决器"""
    
    def detect_conflicts(
        self,
        agent_outputs: Dict[str, Dict[str, Any]]
    ) -> ConflictReport:
        """检测冲突"""
        conflicts = []
        
        # 检查数据一致性
        data_conflicts = self._check_data_consistency(agent_outputs)
        conflicts.extend(data_conflicts)
        
        # 检查方法匹配性
        method_conflicts = self._check_method_consistency(agent_outputs)
        conflicts.extend(method_conflicts)
        
        return ConflictReport(
            report_id=self._generate_id(),
            timestamp=datetime.now(),
            conflict_level=self._determine_level(conflicts),
            conflicts=conflicts,
            source_agents=list(agent_outputs.keys()),
            resolution_status="pending"
        )
    
    def resolve(self, report: ConflictReport) -> bool:
        """解决冲突"""
        for conflict in report.conflicts:
            if conflict.suggested_resolution:
                # 尝试自动解决
                success = self._auto_resolve(conflict)
                if not success:
                    report.resolution_status = "escalated"
                    return False
        
        report.resolution_status = "resolved"
        return True
```

```python
from dataclasses import dataclass
from typing import Dict, Any, Optional

    message: str
    details: Dict[str, Any]
    timestamp: datetime
    source: str
    recoverable: bool = True

            "details": self.details,
            "timestamp": self.timestamp.isoformat(),
            "source": self.source,
            "recoverable": self.recoverable,

class HandoffIncompleteError(MSRAError):

            error_code=ErrorCode.HANDOFF_INCOMPLETE,
            message=f"Handoff 数据缺少必需字段: {missing_fields}",
```

### 23.3.3 错误处理流程

```python
class ErrorHandler:
    """错误处理器"""
    
    def __init__(self):
        self._error_log: List[MSRAError] = []
        self._retry_policies: Dict[ErrorCode, RetryPolicy] = {}
    
    def handle(self, error: MSRAError) -> bool:
        """
        处理错误
        
        Returns:
            bool: 是否可以恢复
        """
        # 记录错误
        self._log_error(error)
        
        # 检查是否可恢复
        if not error.recoverable:
            return False
        
        # 检查重试次数
        if error.retry_count >= error.max_retries:
            return False
        
        # 执行恢复策略
        return self._recover(error)
    
    def _recover(self, error: MSRAError) -> bool:
        """执行恢复策略"""
        recovery_strategies = {
            ErrorCode.GATE_FAILURE: self._recover_gate_failure,
            ErrorCode.SAP_DEVIATION: self._recover_sap_deviation,
            ErrorCode.AGENT_TIMEOUT: self._recover_agent_timeout,
            ErrorCode.CODE_RUNTIME_ERROR: self._recover_code_error,
        }
        
        strategy = recovery_strategies.get(error.error_code)
        if strategy:
            return strategy(error)
        
        return False
    
    def _recover_gate_failure(self, error: MSRAError) -> bool:
        """恢复 Gate 失败"""
        # 尝试重新执行 Gate 检查
        gate_id = error.details.get("gate_id")

        return True
```
